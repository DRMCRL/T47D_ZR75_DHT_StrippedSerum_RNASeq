Pre-Processing Workflow

rm_dot_node <- function(x, node = "all"){
  nd <- x[grepl(node, x)]
  id <- str_replace_all(nd, "\\t([0-9]*).label.+", "\\1")
  pat <- paste(c("(", node, "| ", id, "$)"), collapse = "")
  x[!grepl(pat, x)]
}
add_input_node <- function(x, node = "raw_data", ignore = "(get|build|make|create)", col = "0.00 0.00 0.00", style = "rectangle"){
  # Find the nodes with no input
  all_id <- x %>% 
    str_subset("label") %>% 
    str_replace_all("\\t([0-9]*).label.+", "\\1")
  has_parent <- x %>% 
    str_subset("[0-9]$") %>% 
    str_extract("[0-9]*$")
  to_ignore <- x %>% 
    str_subset(ignore) %>% 
    str_replace_all("\\t([0-9]*).label.+", "\\1")
  no_parent <- setdiff(all_id, c(has_parent, to_ignore))
  # Find the position to insert the node
  new_id <- as.character(max(as.numeric(all_id)) +1)
  new_text <- c(
    "\t", new_id, 
    "[label = \"", node, 
    "\", color = \"", col,
    "\", style=\"", style,
    "\"];"
    ) %>%
    paste(collapse = "")
  last_label <- x %>%
    str_detect("label") %>%
    which() %>%
    max()
  ## Add the node
  x <- c(
    x[seq_len(last_label)], new_text, x[seq(last_label + 1, length(x))]
  )
  new_edges <- paste0("\t", paste(new_id, no_parent, sep = " -> "))
  c(
    x[seq(1, length(x) - 1)],
    new_edges,
    x[length(x)]
  )
}
